useEffect(() => {
  const unsubscribe = subscribe((data) => {
    try {
      console.log("📥 WS DATA:", data); // 🔥 DEBUG

      // Accept BOTH formats:
      // 1. { device_id, reading, risk, health }
      // 2. { device_id, temperature, humidity, ... }

      const device_id = data.device_id;

      if (!device_id) return;

      // Normalize reading
      const reading = data.reading ?? data;

      // Add device if new
      setDevices((prev) => {
        const exists = prev.some((d) => d.device_id === device_id);
        if (exists) return prev;
        return [...prev, { device_id }];
      });

      // Auto-select first device
      setSelectedDeviceId((prev) => prev ?? device_id);

      // ✅ Always update reading
      setLatestReading(reading);

      // Virtual flag (safe fallback)
      setIsVirtual(reading?.is_virtual ?? false);

      // Optional fields
      if (data.risk) setRisk(data.risk);
      if (data.health) setHealth(data.health);

    } catch (err) {
      console.error("❌ DeviceContext error:", err);
    }
  });

  return unsubscribe;
}, [subscribe]);
